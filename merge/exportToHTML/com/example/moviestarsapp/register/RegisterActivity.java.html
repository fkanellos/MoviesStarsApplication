40       protected void onPostCreate(@Nullable Bundle savedInstanceState) {
41           super.onPostCreate(savedInstanceState);
42   
43           mAuth = FirebaseAuth.getInstance();
44   
45           Button button = findViewById(R.id.ok);
46           button.setOnClickListener(RegisterActivity.this);

48           editTextEmail = (EditText)findViewById(R.id.email_edit);

57       @Override
58       public void onClick(View v) {
59           switch (v.getId()){
60               case R.id.ok:
61               registerUser();
62               break;
63           }
64   
65       }
66   
67       private void registerUser() {
68           String email = editTextEmail.getText().toString().trim();
69           String password = editTextPassword.getText().toString().trim();
70           String fullname = editTextFullName.getText().toString().trim();
71           String age = editTextAge.getText().toString().trim();
72   
73           if(fullname.isEmpty()){
74               editTextFullName.setError("Full name is required");
75               editTextFullName.requestFocus();
76               return;
77           }
78           if(email.isEmpty()) {
79               editTextEmail.setError("Email is required");
80               editTextEmail.requestFocus();
81               return;
82           }
83           if(age.isEmpty()){
84               editTextAge.setError("Age is required");
85               editTextAge.requestFocus();
86           }
87           if(!Patterns.EMAIL_ADDRESS.matcher(email).matches()) {
88               editTextEmail.setError("Please provide valid email!");
89               editTextEmail.requestFocus();
90               return;
91           }
92   
93           if(password.isEmpty()){
94              editTextPassword.setError("Password is required!");
95              editTextPassword.requestFocus();
96              return;
97           }
98   
99           if(password.length() < 6){
100             editTextPassword.setError("Min password length should be 6 characters!");
101             editTextPassword.requestFocus();
102             return;
103          }
104  
105          progressbar.setVisibility(View.VISIBLE);
106          mAuth.createUserWithEmailAndPassword(email,password)
107                  .addOnCompleteListener(new OnCompleteListener<AuthResult>() {
108                      @Override
109                      public void onComplete(@NonNull Task<AuthResult> task) {
110                         if(task.isSuccessful()){
111                             User user = new User(fullname, email, age);
112  
113                             FirebaseDatabase.getInstance().getReference("Users")
114                                     .child(FirebaseAuth.getInstance().getCurrentUser().getUid())
115                                     .setValue(user).addOnCompleteListener(new OnCompleteListener<Void>() {
116                                 @Override
117                                 public void onComplete(@NonNull Task<Void> task) {
118  
119                                     if(task.isSuccessful()){
120                                         Toast.makeText(RegisterActivity.this, "User has been registered successfully!", Toast.LENGTH_LONG).show();
121                                         progressbar.setVisibility(View.VISIBLE);
122                                         Intent intent = new Intent(getApplicationContext(), CreateAccountActivity.class);
123                                         startActivity(intent);
124  
125                                     }else{
126                                         Toast.makeText(RegisterActivity.this, "Failed to register! Try again", Toast.LENGTH_LONG).show();
127                                         progressbar.setVisibility(View.GONE);
128                                     }
129                                 }
130                             });
131  
132                         }else{
133                             Toast.makeText(RegisterActivity.this, "Failed to register! Try again", Toast.LENGTH_LONG).show();
134                             progressbar.setVisibility(View.GONE);
135  
136                      }
137                  };
138  
139      });
