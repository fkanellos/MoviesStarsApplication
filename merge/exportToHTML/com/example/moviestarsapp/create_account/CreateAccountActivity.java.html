85       @Override
86       public void onActivityResult(int requestCode, int resultCode, Intent data) {
87           super.onActivityResult(requestCode, resultCode, data);
88   
89           // Result returned from launching the Intent from GoogleSignInApi.getSignInIntent(...);
90           if (requestCode == RC_SIGN_IN) {
91               Task<GoogleSignInAccount> task = GoogleSignIn.getSignedInAccountFromIntent(data);
92               try {
93                   // Google Sign In was successful, authenticate with Firebase
94                   GoogleSignInAccount account = task.getResult(ApiException.class);
95                   firebaseAuthWithGoogle(account);
96               } catch (ApiException e) {
97                   // Google Sign In failed, update UI appropriately
98                   Toast.makeText(this, e.getMessage(), Toast.LENGTH_SHORT).show();
99               }
100          }
101      }

103      private void firebaseAuthWithGoogle(GoogleSignInAccount acct) {
104          AuthCredential credential = GoogleAuthProvider.getCredential(acct.getIdToken(), null);
105          mAuth.signInWithCredential(credential)
106                  .addOnCompleteListener(this, new OnCompleteListener<AuthResult>() {
107                      @Override
108                      public void onComplete(@NonNull Task<AuthResult> task) {
109                          if (task.isSuccessful()) {
110                              // Sign in success
111                              FirebaseUser user = mAuth.getCurrentUser();
112                              Intent intent = new Intent(getApplicationContext(), HomeActivity.class);
113                              startActivity(intent);
114  
115                          } else {
116                              Toast.makeText(CreateAccountActivity.this, "Sorry auth failed.", Toast.LENGTH_SHORT).show();
117                          }
118                      }
119                  });
120  
121      }

143      private void userLogin() {
144          String email = editTextEmail.getText().toString().trim();
145          String password = editTextEmail.getText().toString().trim();
146  
147          if(email.isEmpty()){
148              editTextEmail.setError("Email is required!!");
149              editTextEmail.requestFocus();
150              return;
151          }
152          if(Patterns.EMAIL_ADDRESS.matcher(email).matches()){
153              editTextEmail.setError("Please enter a valid email!!");
154              editTextEmail.requestFocus();
155              return;
156          }
157          if(password.isEmpty()){
158              editTextPassword.setError("Password is required!!");
159              editTextPassword.requestFocus();
160              return;
161          }
162          if(password.length() < 6){
163              editTextPassword.setError("Min password length is 6 characters!!");
164              editTextPassword.requestFocus();
165              return;
166          }
167          progressBar.setVisibility(View.VISIBLE);
168  
169          mAuth.signInWithEmailAndPassword(email, password).addOnCompleteListener(new OnCompleteListener<AuthResult>() {
170              @Override
171              public void onComplete(@NonNull Task<AuthResult> task) {
172                  if(task.isSuccessful()){
173                      FirebaseUser user = FirebaseAuth.getInstance().getCurrentUser();
174                      if(user.isEmailVerified()){
175                          EditText userNameEdit = findViewById(R.id.username_edit);
176                          EditText passwdEdit = findViewById(R.id.password_edit);
177                          String userStr = userNameEdit.getText().toString();
178                          String passwdStr = passwdEdit.getText().toString();
179                          Intent intent = new Intent(CreateAccountActivity.this, HomeActivity.class);
180                          startActivity(intent);
181                      }else {
182                          user.sendEmailVerification();
183                          Toast.makeText(CreateAccountActivity.this, "Check your email to verify your account", Toast.LENGTH_LONG).show();
184                      }
185                      
186                      
187  
188  
189                  }else{
190                      Toast.makeText(CreateAccountActivity.this, "Failed to login! Please check your credentials!", Toast.LENGTH_LONG).show();
191  
192                  }
193              }
194          });
195  
196  
197  
198      }
